50TramaBase* Decodificador::parsearLinea(const char* linea) {
51 if (!linea || miStrlen(linea) < 3) return nullptr;
52
53 char tipo = linea[0];
54
55 // Verificar formato correcto: "L,X" o "M,N"
56 if (linea[1] != ',') return nullptr;
57
58 const char* valorStr = &linea[2]; // Saltar "L," o "M,"
59
60 if (tipo == 'L') {
61 // Trama Load: L,X - tomar primer carácter después de la coma
62 if (miStrlen(valorStr) >= 1) {
63 return new TramaLoad(valorStr[0]);
64 }
65 } else if (tipo == 'M') {
66 // Trama Map: M,N - convertir el resto a int
67 int valor = miAtoi(valorStr);
68 return new TramaMap(valor);
69 }
70
71 return nullptr;
72}
